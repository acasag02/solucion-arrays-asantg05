001    package es.unileon.prg.tema7;
002    /**
003     * Clase que representa un individuo.
004     *
005     * @author MTP
006     * @version 3.0
007     */
008    public class Persona {
009        /**
010         * Nombre del individuo.
011         */
012        private String _nombre;
013    
014        /**
015         * Telefono del individuo.
016         */
017        private String _telefono;
018    
019        /**
020         * Constructor que recibe todos los datos necesarios de una persona.
021         *
022         * @param Nombre
023         *            Nombre del individuo
024         * @param telefono
025         *            Telefono del individuo
026         */
027        public Persona(String nombre, String telefono) {
028            _nombre = nombre;
029            _telefono = telefono;
030        }
031    
032        /**
033         * Constructor de copia. Crear una personas a partir de otra recibida como
034         * argumento.
035         *
036         * @param otra
037         *            Individuo a copiar.
038         */
039        public Persona(Persona otra) {
040            _nombre = otra.obtenerNombre();
041            _telefono = otra.obtenerTelefono();
042        }
043    
044        /**
045         * Devuelve el nombre del individuo
046         *
047         * @return Nombre del individuo
048         */
049        public String obtenerNombre() {
050            return _nombre;
051        }
052    
053        /**
054         * Devuelve el telefono del individuo
055         *
056         * @return Telefono del individuo
057         */
058        public String obtenerTelefono() {
059            return _telefono;
060        }
061    
062        /**
063         * Dos personas se consideran igual si tienen el mismo nombre y numero de
064         * telefono.
065         *
066         * @param otra
067         *            Individuo a comparar.
068         * @return <code> verdadero </code> Son la misma persona.
069         *         <code> falso </code> En caso contrario
070         */
071        public boolean esIgual(Persona otra) {
072            return (_nombre.equals(otra.obtenerNombre()) &&
073                        _telefono.equals(otra.obtenerTelefono()));
074        }
075    
076    
077            /*
078             * (non-Javadoc)
079             * @see java.lang.Object#toString()
080             */
081            public String toString() {
082                    StringBuffer salida = new StringBuffer();
083    
084                    salida.append("Nombre: "+_nombre+" ");
085                    salida.append("Telefono: "+_telefono+" ");
086    
087                    return salida.toString();
088            }
089    }
