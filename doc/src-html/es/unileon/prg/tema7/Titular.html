001    package es.unileon.prg.tema7;
002    /**
003     * Clase que representa un titular de una cuenta bancaria
004     *
005     * @author MTP
006     * @version 3.0
007     */
008    public class Titular{
009        /**
010         * Nif del titular
011         */
012        private String _nif;
013        /**
014         * Nombre del titular
015         */
016        private String _nombre;
017        /**
018         * Edad del titular
019         */
020        private int _edad;
021        /**
022         * Cantidad que le permite el banco tener al descubierto
023         */
024        private int _credito;
025    
026        /**
027         * Constructor que recibe todos los datos necesarios de una persona
028         * @param nif Nif del titular
029         * @param nombre Nombre del titular
030         * @param edad Edad del titular
031         * @param direccion Domicilio del titular
032         * @param credito Cantidad que le permite el banco tener al descubierto
033         */
034        public Titular (String nif,
035                 String nombre,
036                 int edad,
037                 int credito){
038                             _nif = new String(nif);
039                             _nombre = new String(nombre);
040                             _edad = edad;
041                             _credito = credito;
042        }
043    
044        /**
045         * Devuelve el nif del titular
046         *
047         * @return el nif del titular
048         */
049        public String obtenerNif (){
050            return _nif;
051        }
052    
053        /**
054         * Devuelve el nombre del titular
055         *
056         * @return el nombre del titular
057         */
058        public String obtenerNombre (){
059            return _nombre;
060        }
061    
062        /**
063         * Devuelve la edad del titular
064         *
065         * @return la edad del titular
066         */
067        public int obtenerEdad (){
068            return _edad;
069        }
070    
071        /**
072         * Devuelve la credito del titular
073         *
074         * @return la credito del titular
075         */
076        public int obtenerCredito (){
077            return _credito;
078        }
079    
080        /**
081         * Comprueba si el titular es mayor de edad
082         * @return verdadero si el titular es mayor de edad y falso en
083         * caso contrario
084         */
085        public boolean esMayorEdad(){
086            if (_edad < 18)
087                return false;
088            else
089                return true;
090        }
091    
092            /*
093             * (non-Javadoc)
094             * @see java.lang.Object#toString()
095             */
096        public String toString(){
097                    StringBuffer salida = new StringBuffer();
098    
099                    salida.append("\n");
100                    salida.append("NIF      : " + _nif + "\n");
101                    salida.append("Nombre   : " + _nombre + "\n");
102                    salida.append("Edad     : " + _edad + "\n");
103                    salida.append("Credito  : " + _credito + "\n");
104    
105                    return salida.toString();
106        }
107    }
