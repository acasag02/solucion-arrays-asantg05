001    package es.unileon.prg.tema7;
002    /**
003     * Clase que representa un banco
004     *
005     * @author MTP
006     * @version 3.0
007     */
008    public class Banco{
009    
010            /**
011             * Numero maximo de titulares de la cuenta
012             */
013            private static int NUM_MAX_CUENTAS = 50;
014    
015        /**
016         * Nombre del banco
017         */
018        private String _nombre;
019        /**
020         * Cuentas bancaria del banco
021         */
022        private CuentaBancaria[] _cuentas;
023    
024        /**
025         * Constructor que recibe por parámetro el nombre del banco
026         *
027         * @param nombre Nomre del banco
028         */
029        public Banco(String nombre){
030                    _nombre = nombre;
031                    _cuentas = new CuentaBancaria[NUM_MAX_CUENTAS];
032        }
033    
034        /**
035         * Se añade una cuenta al banco
036         *
037         * @return <code> true </code> Proceso correcto <code> false </code> En caso contrario
038         */
039        public boolean anyadir(CuentaBancaria cuenta) {
040                    boolean correcto;
041    
042                    int contador = 0;
043                    correcto = false;
044                    while ((contador < _cuentas.length) && (_cuentas[contador] != null)) {
045                            contador++;
046                    }
047    
048                    if(contador != NUM_MAX_CUENTAS) {
049                            _cuentas[contador] = cuenta;
050                            correcto = true;
051                    }
052    
053                    return correcto;
054            }
055    
056            /*
057             * (non-Javadoc)
058             * @see java.lang.Object#toString()
059             */
060        public String toString(){
061                    StringBuffer salida = new StringBuffer();
062    
063                    salida.append("Banco  : " + _nombre + "\n");
064                    int contador = 0;
065                    while ((contador < _cuentas.length) && (_cuentas[contador] != null)) {
066                            salida.append("\n");
067                            salida.append("Cuenta " + (contador+1) + " ");
068                            salida.append(_cuentas[contador]);
069                            contador++;
070                    }
071    
072                    return salida.toString();
073        }
074    }
