001    package es.unileon.prg.tema7;
002    /**
003     * Clase que representa una cuenta bancaria
004     *
005     * @author mtp
006     * @version 3.0
007     */
008    public class CuentaBancaria{
009    
010            /**
011             * Numero maximo de titulares de la cuenta
012             */
013            private static int NUM_MAX_TITULARES = 5;
014    
015        /**
016         * Saldo de la cuenta
017         */
018        private int _saldo;
019        /**
020         * Titulares de la cuenta
021         *
022         * @see Titular
023         */
024        private Titular[] _titulares;
025    
026        /**
027         * Constructor que recibe por parámetro todos los datos necesarios
028         * para crear un objeto CuentaBancaria
029         *
030         * @param saldo Saldo inicial de la cuenta
031         * @param titular Titular principal de la cuenta
032         */
033        public CuentaBancaria (int saldo, Titular titular){
034            _saldo = saldo;
035            _titulares = new Titular[NUM_MAX_TITULARES];
036            _titulares[0] = titular;
037        }
038    
039        /**
040         * Devuelve el saldo de la cuenta
041         *
042         * @return valor del saldo de la cuenta
043         */
044        public int obtenerSaldo (){
045            return _saldo;
046        }
047    
048        /**
049         * Devuelve el titular principal de la cuenta. Se toma como
050         * titular principal el primero de la lista.
051         *
052         * @return Titular principal de la cuenta
053         */
054        public Titular obtenerTitularPrincipal() {
055                    return _titulares[0];
056            }
057    
058        /**
059         * Devuelve el credito del titular principal
060         *
061         * @return valor del credito del titular principal
062         */
063        int obtenerCredito (){
064            return obtenerTitularPrincipal().obtenerCredito();
065        }
066    
067        /**
068         * Se añade un titular a la cuenta
069         *
070         * @return <code> true </code> Proceso correcto <code> false </code> En caso contrario
071         */
072        public boolean anyadir(Titular titular) {
073                    boolean correcto;
074    
075                    int contador = 0;
076                    correcto = false;
077                    while ((contador < _titulares.length) && (_titulares[contador] != null)) {
078                            contador++;
079                    }
080    
081                    if(contador != NUM_MAX_TITULARES) {
082                            _titulares[contador] = titular;
083                            correcto = true;
084                    }
085    
086                    return correcto;
087            }
088    
089        /**
090         * Permite ingresar una determinada cantidad en la cuenta
091         *
092         * @param cantidad dinero a ingresar
093         */
094        public void ingresar (int cantidad){
095    
096            System.out.println("Vamos a ingresar interes por valor " + cantidad);
097            System.out.println("Vamos a ingresar " + cantidad + " ?...");
098            _saldo = _saldo + cantidad;
099            System.out.println("...ya hemos ingresado " + cantidad + " ?"+ _saldo);
100        }
101    
102        /**
103         * Permite sacar una determinada cantidad de la cuenta, siempre y
104         * cuando el titular principal de la cuenta sea mayor de edad y el saldo
105         * más el crédito del titular principal sea suficiente para esa cantidad
106         *
107         * @param cantidad dinero a sacar
108         */
109        public void sacar (int cantidad){
110                    if ( !obtenerTitularPrincipal().esMayorEdad() ){
111                            System.out.println("Los menores de edad no pueden sacar dinero.");
112                    } else {
113                            if ( _saldo + obtenerCredito() < cantidad ){
114                                    System.out.println("Operacion cancelada: credito insuficiente.");
115                            } else {
116                                    System.out.println("Vamos a sacar " + cantidad + " ?...");
117                                    _saldo = _saldo - cantidad;
118                                    System.out.println("...ya hemos sacado " + cantidad +
119                                       " ? " + _saldo);
120                        }
121                    }
122        }
123    
124            /*
125             * (non-Javadoc)
126             * @see java.lang.Object#toString()
127             */
128        public String toString(){
129                    StringBuffer salida = new StringBuffer();
130    
131                    salida.append("\n");
132                    salida.append("Saldo  : " + _saldo + "\n");
133    
134                    int contador = 0;
135                    while ((contador < _titulares.length) && (_titulares[contador] != null)) {
136                            salida.append(_titulares[contador]);
137                            contador++;
138                    }
139    
140                    return salida.toString();
141        }
142    }
